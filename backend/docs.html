Organizations

/api/organizations

Fields

name *

String (100 chars)

location

String (255 chars)

logo_url

String (255 chars)

currency

String (default: 'KES')

GET /api/organizations

Retrieve all organizations

// Response
[
  {
    "id": 1,
    "name": "Acme Corp",
    "location": "Nairobi, Kenya",
    "logo_url": "https://...",
    "currency": "KES",
    "created_at": "2023-07-20T10:00:00Z",
    "updated_at": "2023-07-20T10:00:00Z"
  }
]
POST /api/organizations

Create a new organization

// Request Body
{
  "name": "New Org",
  "location": "Mombasa, Kenya",
  "logo_url": "https://...",
  "currency": "KES"
}
GET /api/organizations/{id}

Retrieve a specific organization

PUT/PATCH /api/organizations/{id}

Update an organization

// Request Body
{
  "name": "Updated Org Name",
  "location": "Updated Location"
}
DELETE /api/organizations/{id}

Delete an organization

Fields

Integer

String (50 chars, unique)

String (255 chars)

String (100 chars, unique)

Enum: employee, admin, super_admin (default: employee)

Retrieve all users (admin only)

{
  "organization_id": 1,
  "username": "newuser",
  "password": "securepassword",
  "email": "user@example.com",
  "user_type": "employee"
}

PUT/PATCH /api/users/{id}

Update a user

DELETE /api/users/{id}

Pay Runs

/api/organizations/{id}/payruns

Fields

organization_id *

Integer

payrun_name *

String (100 chars)

pay_period_start *

Date

pay_period_end *

Date

pay_frequency

Enum: weekly, bi-weekly, monthly (default: monthly)

status

Enum: draft, reviewed, finalized (default: draft)

created_by *

Integer (user ID)

reviewed_by

Integer (user ID)

finalized_by

Integer (user ID)

GET /api/organizations/{id}/payruns

Retrieve all pay runs for an organization

POST /api/organizations/{id}/payruns

Create a new pay run

// Request Body
{
  "payrun_name": "July 2023 Payroll",
  "pay_period_start": "2023-07-01",
  "pay_period_end": "2023-07-31",
  "pay_frequency": "monthly",
  "status": "draft"
}
GET /api/organizations/{org_id}/payruns/{payrun_id}

Retrieve a specific pay run

PUT/PATCH /api/organizations/{org_id}/payruns/{payrun_id}

Update a pay run

// Request Body (example: finalizing a pay run)
{
  "status": "finalized",
  "finalized_by": 1
}
DELETE /api/organizations/{org_id}/payruns/{payrun_id}

Delete a pay run (only allowed for draft status)

GET /api/payruns/{id}/details

// Request Body

  "employee_id": 1,

  "overtime_amount": 5000.00,

  "net_pay": 47500.00
}

DELETE /api/payruns/{payrun_id}/details/{detail_id}
